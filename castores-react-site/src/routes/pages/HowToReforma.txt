import { useNavigate } from "react-router-dom";
import { useDarkMode } from "../../hooks/contex/DarkModeContex";
import { useLangContex } from "../../hooks/contex/LangContex";
import { TITLE } from "../../helpers/configs";
import { Typography, Button } from "@mui/material";
import HowToCard from "./Reforma/HowToCard";
import ArrowBackRoundedIcon from "@mui/icons-material/ArrowBackRounded";

// images
import t1 from "../../static/img/reforma/howTo/t1.webp";
import t2 from "../../static/img/reforma/howTo/t2.webp";
import t3 from "../../static/img/reforma/howTo/t3.webp";

const HowToReforma = () => {
    document.title = TITLE + " | How to";
    const navigate = useNavigate();
    const { themeTatailwind } = useDarkMode();
    const { lang } = useLangContex();

    const dataHow = [
        {
            title: lang.how_to_use.card1.title,
            md: lang.how_to_use.card1.content,
            image: t1,
        },
        {
            title: lang.how_to_use.card2.title,
            md: lang.how_to_use.card2.content,
            image: t2,
        },
        {
            title: lang.how_to_use.card3.title,
            md: lang.how_to_use.card3.content,
            image: t3,
        },
    ];

    return (
        <div className="flex flex-col items-center m-5">
            <div className="relative flex max-w-7xl w-full">
                <div className="absolute z-0 w-full h-full max-w-5xl top-0 left-0 right-0 bottom-0 m-auto reformaSection rounded-2xl blur-2xl drop-shadow-xs"></div>
                <Typography
                    className="flex justify-center w-full z-10"
                    color={themeTatailwind.primary.color}
                    variant="h4"
                >
                    <b>{lang.how_to_use.title}</b>
                </Typography>
            </div>
            <div className="flex justify-start max-w-7xl w-full mt-5 z-10">
                <Button
                    variant="contained"
                    color="primary"
                    startIcon={<ArrowBackRoundedIcon />}
                    onClick={() => navigate("/")}
                >
                    {lang.how_to_use.back_btn}
                </Button>
            </div>

            {dataHow.map((data, index) => {
                return (
                    <HowToCard
                        key={index}
                        title={data.title}
                        description={data.md}
                        image={data.image}
                    />
                );
            })}
        </div>
    );
};

export default HowToReforma;
